Class RadarSeriesStyles
 
Packageseries
Classpublic class RadarSeries

RadarChart 컨트롤의 series property에 할당되는 데이터 시리즈 입니다.

데이터를 바탕으로 실제적인 표현하는 클래스입니다.

레이아웃은 다음과 같은 형식으로 작성됩니다.

 <rMateChart backgroundColor="0xFFFFEE" cornerRadius="12" borderStyle="solid">
        <RadarChart isSeriesOnAxis="false" type="polygon" paddingTop="20" paddingBottom="20">
             <radialAxisRenderers>
               <AxisRenderer horizontal="true" visible="true"/>
               <AxisRenderer horizontal="false" visible="false"/>
              </radialAxisRenderers>
              <radialAxis>
               <LinearAxis/>
              </radialAxis>
              <angularAxis>
                  <CategoryAxis id="aAxis" categoryField="catName" displayName="Category" />
              </angularAxis>
              <series>
                  <RadarSeries field="year2005" radius="5" displayName="2005년"/>
                  <RadarSeries field="year2006" radius="5" displayName="2006년"/>
                  <RadarSeries field="year2007" radius="5" displayName="2007년"/>
              </series>
   </RadarChart>
 </rMateChart>
